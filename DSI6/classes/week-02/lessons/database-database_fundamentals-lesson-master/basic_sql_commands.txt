# SELECT

SELECT *
FROM houses
WHERE age>23


SELECT *
FROM houses
WHERE age>23 AND price>10000


SELECT *
FROM houses
WHERE age<30 OR bdrms>3


SELECT bdrms, avg(price)
FROM houses
GROUP BY bdrms


SELECT age,bdrms, avg(price)
FROM houses
GROUP BY age,bdrms
ORDER BY age,bdrms
